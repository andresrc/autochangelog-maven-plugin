163     public void execute() throws MojoExecutionException, MojoFailureException {
164         getLog().debug("Looking for previous release of " + project.getGroupId() + ":" + project.getArtifactId() + ":"
165                 + project.getVersion());
166         Artifact projectArtifact = artifactFactory
167                 .createProjectArtifact(project.getGroupId(), project.getArtifactId(), project.getVersion());
168         ArtifactVersion projectVersion = new DefaultArtifactVersion(project.getVersion());
169 
170         ArtifactVersion latest = null;
171         try {
172             List<ArtifactVersion> artifactVersions = artifactMetadataSource
173                     .retrieveAvailableVersions(projectArtifact, localRepository,
174                             project.getRemoteArtifactRepositories());
175             for (ArtifactVersion version : artifactVersions) {
176                 if (SNAPSHOT_PATTERN.matcher(version.toString()).find() || projectVersion.compareTo(version) <= 0) {
177                     continue;
178                 }
179                 if (latest == null || latest.compareTo(version) < 0) {
180                     latest = version;
181                 }
182             }
183         } catch (ArtifactMetadataRetrievalException e) {
184             throw new MojoExecutionException(e.getMessage(), e);
185         }
186         getLog().debug("Previous release = " + latest);
187 
188         Map<GroupArtifactId, String> addedDeps = new LinkedHashMap<GroupArtifactId, String>();
189         Map<GroupArtifactId, String> removedDeps = new LinkedHashMap<GroupArtifactId, String>();
190         Map<GroupArtifactId, Map.Entry<String, String>> updatedDeps =
191                 new LinkedHashMap<GroupArtifactId, Map.Entry<String, String>>();
192 
193         for (Dependency dep : project.getDependencies()) {
194             addedDeps.put(new GroupArtifactId(dep), dep.getVersion());
195         }
196 
197         String startTag = null;
198         String endTag = null;
199         if (latest != null) {
200             Artifact latestArtifact = artifactFactory
201                     .createProjectArtifact(project.getGroupId(), project.getArtifactId(), latest.toString());
202             try {
203                 artifactResolver.resolve(latestArtifact, project.getRemoteArtifactRepositories(), localRepository);
204             } catch (ArtifactResolutionException e) {
205                 throw new MojoExecutionException(e.getMessage(), e);
206             } catch (ArtifactNotFoundException e) {
207                 throw new MojoExecutionException(e.getMessage(), e);
208             }
209 
210             ProjectBuildingRequest request = new DefaultProjectBuildingRequest();
211 
212             request.setProcessPlugins(true);
213             request.setProfiles(request.getProfiles());
214             request.setActiveProfileIds(session.getRequest().getActiveProfiles());
215             request.setInactiveProfileIds(session.getRequest().getInactiveProfiles());
216             request.setRemoteRepositories(session.getRequest().getRemoteRepositories());
217             request.setSystemProperties(session.getSystemProperties());
218             request.setUserProperties(session.getUserProperties());
219             request.setRemoteRepositories(session.getRequest().getRemoteRepositories());
220             request.setPluginArtifactRepositories(session.getRequest().getPluginArtifactRepositories());
221             request.setRepositorySession(session.getRepositorySession());
222             request.setLocalRepository(localRepository);
223             request.setBuildStartTime(session.getRequest().getStartTime());
224             request.setResolveDependencies(true);
225             request.setValidationLevel(ModelBuildingRequest.VALIDATION_LEVEL_STRICT);
226             MavenProject latestProject;
227             try {
228                 latestProject = projectBuilder.build(latestArtifact.getFile(), request).getProject();
229             } catch (ProjectBuildingException e) {
230                 throw new MojoExecutionException(e.getMessage(), e);
231             }
232             for (Dependency dep : latestProject.getDependencies()) {
233                 GroupArtifactId key = new GroupArtifactId(dep);
234                 if (addedDeps.containsKey(key)) {
235                     String v = addedDeps.get(key);
236                     addedDeps.remove(key);
237                     if (!v.equals(dep.getVersion())) {
238                         updatedDeps.put(key, new AbstractMap.SimpleImmutableEntry<String, String>(dep.getVersion(), v));
239                     }
240                 } else if (updatedDeps.containsKey(key)) {
241                     String v = updatedDeps.get(key).getValue();
242                     if (!v.equals(dep.getVersion())) {
243                         updatedDeps.remove(key);
244                         addedDeps.put(key, v);
245                     } else {
246                         updatedDeps.put(key, new AbstractMap.SimpleImmutableEntry<String, String>(dep.getVersion(), v));
247                     }
248                 } else {
249                     removedDeps.put(key, dep.getVersion());
250                 }
251             }
252 
253             String tagNameFormat = getTagNameFormat(latestProject);
254             getLog().debug("Start Tag name format = " + tagNameFormat);
255             Interpolator interpolator = new StringSearchInterpolator("@{", "}");
256             List<String> possiblePrefixes = java.util.Arrays.asList("project", "pom");
257             Properties values = new Properties();
258             values.setProperty("artifactId", project.getArtifactId());
259             values.setProperty("groupId", project.getGroupId());
260             values.setProperty("version", latest.toString());
261             interpolator.addValueSource(new PrefixedPropertiesValueSource(possiblePrefixes, values, true));
262             RecursionInterceptor recursionInterceptor = new PrefixAwareRecursionInterceptor(possiblePrefixes);
263             try {
264                 startTag = interpolator.interpolate(tagNameFormat, recursionInterceptor);
265             } catch (InterpolationException e) {
266                 throw new MojoExecutionException("Could not interpolate specified tag name format: " + tagNameFormat,
267                         e);
268             }
269         }
270         if (!SNAPSHOT_PATTERN.matcher(project.getVersion()).find()) {
271             String tagNameFormat = getTagNameFormat(project);
272             getLog().debug("End Tag name format = " + tagNameFormat);
273             Interpolator interpolator = new StringSearchInterpolator("@{", "}");
274             List<String> possiblePrefixes = java.util.Arrays.asList("project", "pom");
275             Properties values = new Properties();
276             values.setProperty("artifactId", project.getArtifactId());
277             values.setProperty("groupId", project.getGroupId());
278             values.setProperty("version", project.getVersion());
279             interpolator.addValueSource(new PrefixedPropertiesValueSource(possiblePrefixes, values, true));
280             RecursionInterceptor recursionInterceptor = new PrefixAwareRecursionInterceptor(possiblePrefixes);
281             try {
282                 endTag = interpolator.interpolate(tagNameFormat, recursionInterceptor);
283             } catch (InterpolationException e) {
284                 throw new MojoExecutionException("Could not interpolate specified tag name format: " + tagNameFormat,
285                         e);
286             }
287         }
288         getLog().info("Start tag = " + startTag);
289         getLog().info("End tag = " + endTag);
290 
291         ChangeLogSet changeLog = null;
292 
293         ScmRepository repository;
294         try {
295             repository = getScmRepository();
296             ScmProvider provider = null;
297             try {
298                 provider = manager.getProviderByRepository(repository);
299             } catch (NoSuchScmProviderException e) {
300                 throw new MojoExecutionException("Unknown/unsupported SCM provider", e);
301             }
302 
303             ScmVersion scmStartVersion;
304             ScmVersion scmEndVersion;
305             if (repository.getProvider().equals("svn")) {
306                 getLog().warn("SVN does not support the required changelog operations");
307             } else {
308                 scmStartVersion = startTag == null ? null : new ScmTag(startTag);
309                 scmEndVersion = endTag == null ? null : new ScmTag(endTag);
310                 ChangeLogScmResult changeLogScmResult;
311                 try {
312                     changeLogScmResult =
313                             provider.changeLog(repository, new ScmFileSet(basedir), scmStartVersion, scmEndVersion);
314                 } catch (ScmException e) {
315                     throw new MojoExecutionException("Could not fetch changelog", e);
316                 }
317                 checkResult(changeLogScmResult);
318 
319                 changeLog = changeLogScmResult.getChangeLog();
320             }
321         } catch (ScmException e) {
322             getLog().error("Could not get SCM repository", e);
323         }
324 
325         if (StringUtils.isEmpty(encoding)) {
326             getLog().warn("Output file encoding has not been set, using platform encoding " + System
327                     .getProperty("file.encoding") + ", i.e. build is platform dependent!");
328             encoding = System.getProperty("file.encoding");
329         }
330 
331         if (!outputDirectory.isDirectory() && !outputDirectory.mkdirs()) {
332             throw new MojoExecutionException("Could not create directory " + outputDirectory);
333         }
334 
335         File outputFile = new File(outputDirectory, finalName + "-changes.json");
336 
337         JsonFactory factory = new JsonFactory();
338         try {
339             JsonGenerator generator = factory.createJsonGenerator(outputFile, JsonEncoding.UTF8);
340             generator.setPrettyPrinter(new DefaultPrettyPrinter());
341             try {
342                 generator.writeStartObject();
343                 generator.writeFieldName("dependencies");
344                 writeDependenciesDelta(addedDeps, removedDeps, updatedDeps, generator);
345                 if (changeLog != null) {
346                     generator.writeFieldName("scm");
347                     writeChangeLog(generator, changeLog);
348                 }
349                 generator.writeEndObject();
350             } finally {
351                 generator.close();
352             }
353         } catch (IOException e) {
354             throw new MojoExecutionException(e.getMessage(), e);
355         }
356         projectHelper.attachArtifact(project, "json", "changes", outputFile);
357     }

359     private void writeDependenciesDelta(Map<GroupArtifactId, String> addedDeps,
360                                         Map<GroupArtifactId, String> removedDeps,
361                                         Map<GroupArtifactId, Map.Entry<String, String>> updatedDeps,
362                                         JsonGenerator generator) throws IOException {
363         generator.writeStartObject();
364         if (!addedDeps.isEmpty()) {
365             generator.writeArrayFieldStart("added");
366             for (Map.Entry<GroupArtifactId, String> e : addedDeps.entrySet()) {
367                 generator.writeStartObject();
368                 generator.writeStringField("groupId", e.getKey().getGroupId());
369                 generator.writeStringField("artifactId", e.getKey().getArtifactId());
370                 generator.writeStringField("version", e.getValue());
371                 generator.writeEndObject();
372             }
373             generator.writeEndArray();
374         }
375         if (!removedDeps.isEmpty()) {
376             generator.writeArrayFieldStart("removed");
377             for (Map.Entry<GroupArtifactId, String> e : removedDeps.entrySet()) {
378                 generator.writeStartObject();
379                 generator.writeStringField("groupId", e.getKey().getGroupId());
380                 generator.writeStringField("artifactId", e.getKey().getArtifactId());
381                 generator.writeStringField("version", e.getValue());
382                 generator.writeEndObject();
383             }
384             generator.writeEndArray();
385         }
386         if (!updatedDeps.isEmpty()) {
387             generator.writeArrayFieldStart("updated");
388             for (Map.Entry<GroupArtifactId, Map.Entry<String, String>> e : updatedDeps.entrySet()) {
389                 generator.writeStartObject();
390                 generator.writeStringField("groupId", e.getKey().getGroupId());
391                 generator.writeStringField("artifactId", e.getKey().getArtifactId());
392                 generator.writeStringField("version", e.getValue().getValue());
393                 generator.writeStringField("originalVersion", e.getValue().getKey());
394                 generator.writeEndObject();
395             }
396             generator.writeEndArray();
397         }
398         generator.writeEndObject();
399     }

401     private String getTagNameFormat(MavenProject project) {
402         Plugin releasePlugin = null;
403         for (Plugin plugin : project.getBuild().getPlugins()) {
404             if ("org.apache.maven.plugins".equals(plugin.getGroupId()) && "maven-release-plugin"
405                     .equals(plugin.getArtifactId())) {
406                 releasePlugin = plugin;
407                 break;
408             }
409         }
410         String tagNameFormat = "@{project.artifactId}-@{project.version}";
411         if (releasePlugin != null) {
412             Xpp3Dom dom = (Xpp3Dom) releasePlugin.getConfiguration();
413             Xpp3Dom tagNameFormatDom = dom.getChild("tagNameFormat");
414             if (tagNameFormatDom != null) {
415                 tagNameFormat = tagNameFormatDom.getValue();
416             }
417         }
418         return tagNameFormat;
419     }

421     private void writeChangeLog(JsonGenerator generator, ChangeLogSet changeLog) throws IOException {
422         generator.writeStartArray();

424             for (ChangeSet changeSet : changeLog.getChangeSets()) {

427         } finally {
428             generator.writeEndArray();
429         }

432     private void writeChangeSet(JsonGenerator generator, ChangeSet changeSet) throws IOException {
433         generator.writeStartObject();
434         try {
435             if (StringUtils.isNotEmpty(changeSet.getAuthor())) {
436                 generator.writeStringField("author", changeSet.getAuthor());
437             }
438             if (StringUtils.isNotEmpty(changeSet.getComment())) {
439                 generator.writeStringField("comment", changeSet.getComment());
440             }
441             if (changeSet.getDate() != null) {
442                 SimpleDateFormat iso8601format = new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ss'Z'");
443                 iso8601format.setTimeZone(TimeZone.getTimeZone("UTC"));
444                 generator.writeStringField("date", iso8601format.format(changeSet.getDate()));
445             }
446             if (changeSet.getParentRevision() != null) {
447                 generator.writeStringField("parentRevision", changeSet.getParentRevision());
448             }
449             if (!changeSet.getMergedRevisions().isEmpty()) {
450                 generator.writeArrayFieldStart("mergedRevisions");
451                 try {
452                     for (String revision : changeSet.getMergedRevisions()) {
453                         generator.writeString(revision);
454                     }
455                 } finally {
456                     generator.writeEndArray();
457                 }
458             }
459             if (!changeSet.getFiles().isEmpty()) {
460                 generator.writeArrayFieldStart("files");
461                 try {
462                     for (ChangeFile file : changeSet.getFiles()) {
463                         writeChangeFile(generator, file);
464                     }
465                 } finally {
466                     generator.writeEndArray();
467                 }
468             }
469         } finally {
470             generator.writeEndObject();
471         }
472     }

474     private void writeChangeFile(JsonGenerator generator, ChangeFile file) throws IOException {
475         generator.writeStartObject();
476         try {
477             if (file.getAction() != null) {
478                 generator.writeStringField("action", file.getAction().toString());
479             }
480             if (file.getName() != null) {
481                 generator.writeStringField("name", file.getName());
482             }
483             if (file.getRevision() != null) {
484                 generator.writeStringField("revision", file.getRevision());
485             }
486             if (file.getOriginalName() != null) {
487                 generator.writeStringField("originalName", file.getOriginalName());
488             }
489             if (file.getOriginalRevision() != null) {
490                 generator.writeStringField("originalRevision", file.getOriginalRevision());
491             }
492         } finally {
493             generator.writeEndObject();
494         }

497     public ScmRepository getScmRepository()
498             throws ScmException {
499         ScmRepository repository;
500 
501         try {
502             repository = manager.makeScmRepository(getConnection());
503 
504             ScmProviderRepository providerRepo = repository.getProviderRepository();
505 
506             if (!StringUtils.isEmpty(username)) {
507                 providerRepo.setUser(username);
508             }
509 
510             if (!StringUtils.isEmpty(password)) {
511                 providerRepo.setPassword(password);
512             }
513 
514             if (repository.getProviderRepository() instanceof ScmProviderRepositoryWithHost) {
515                 ScmProviderRepositoryWithHost repo = (ScmProviderRepositoryWithHost) repository.getProviderRepository();
516 
517                 loadInfosFromSettings(repo);
518 
519                 if (!StringUtils.isEmpty(username)) {
520                     repo.setUser(username);
521                 }
522 
523                 if (!StringUtils.isEmpty(password)) {
524                     repo.setPassword(password);
525                 }
526 
527                 if (!StringUtils.isEmpty(privateKey)) {
528                     repo.setPrivateKey(privateKey);
529                 }
530 
531                 if (!StringUtils.isEmpty(passphrase)) {
532                     repo.setPassphrase(passphrase);
533                 }
534             }
535 
536             if (!StringUtils.isEmpty(tagBase) && repository.getProvider().equals("svn")) {
537                 SvnScmProviderRepository svnRepo = (SvnScmProviderRepository) repository.getProviderRepository();
538 
539                 svnRepo.setTagBase(tagBase);
540             }
541         } catch (Exception e) {
542             throw new ScmException("Can't load the scm provider.", e);
543         }
544 
545         return repository;
546     }

548     /**
549      * Load username password from settings if user has not set them in JVM properties
550      *
551      * @param repo
552      */
553     private void loadInfosFromSettings(ScmProviderRepositoryWithHost repo) {
554         if (username == null || password == null) {
555             String host = repo.getHost();
556 
557             int port = repo.getPort();
558 
559             if (port > 0) {
560                 host += ":" + port;
561             }
562 
563             Server server = this.settings.getServer(host);
564 
565             if (server != null) {
566                 if (username == null) {
567                     username = this.settings.getServer(host).getUsername();
568                 }
569 
570                 if (password == null) {
571                     password = this.settings.getServer(host).getPassword();
572                 }
573 
574                 if (privateKey == null) {
575                     privateKey = this.settings.getServer(host).getPrivateKey();
576                 }
577 
578                 if (passphrase == null) {
579                     passphrase = this.settings.getServer(host).getPassphrase();
580                 }
581             }
582         }
583     }

600     /**
601      * used to retrieve the SCM connection string
602      *
603      * @return the url string used to connect to the SCM
604      * @throws MojoExecutionException when there is insufficient information to retrieve the SCM connection string
605      */
606     protected String getConnection()
607             throws MojoExecutionException {
608         if (this.connection != null) {
609             return connection;
610         }
611 
612         if (project.getScm() == null) {
613             throw new MojoExecutionException("SCM Connection is not set.");
614         }
615 
616         String scmConnection = project.getScm().getConnection();
617         if (StringUtils.isNotEmpty(scmConnection) && "connection".equals(connectionType.toLowerCase())) {
618             connection = scmConnection;
619         }
620 
621         String scmDeveloper = project.getScm().getDeveloperConnection();
622         if (StringUtils.isNotEmpty(scmDeveloper) && "developerconnection".equals(connectionType.toLowerCase())) {
623             connection = scmDeveloper;
624         }
625 
626         if (StringUtils.isEmpty(connection)) {
627             throw new MojoExecutionException("SCM Connection is not set.");
628         }
629 
630         return connection;
631     }

633     /**
634      * Determines the relative path from trunk to tag, and adds this relative path
635      * to the url.
636      *
637      * @param trunkPath - The trunk url
638      * @param tagPath   - The tag base
639      * @param urlPath   - scm.url or scm.connection
640      * @return The url path for the tag.
641      */
642     private String translateUrlPath(String trunkPath, String tagPath, String urlPath) {
643         trunkPath = trunkPath.trim();
644         tagPath = tagPath.trim();
645         //Strip the slash at the end if one is present
646         if (trunkPath.endsWith("/")) {
647             trunkPath = trunkPath.substring(0, trunkPath.length() - 1);
648         }
649         if (tagPath.endsWith("/")) {
650             tagPath = tagPath.substring(0, tagPath.length() - 1);
651         }
652         char[] tagPathChars = trunkPath.toCharArray();
653         char[] trunkPathChars = tagPath.toCharArray();
654         // Find the common path between trunk and tags
655         int i = 0;
656         while ((i < tagPathChars.length) && (i < trunkPathChars.length) && tagPathChars[i] == trunkPathChars[i]) {
657             ++i;
658         }
659         // If there is nothing common between trunk and tags, or the relative
660         // path does not exist in the url, then just return the tag.
661         if (i == 0 || urlPath.indexOf(trunkPath.substring(i)) < 0) {
662             return tagPath;
663         } else {
664             return StringUtils.replace(urlPath, trunkPath.substring(i), tagPath.substring(i));
665         }
666     }
