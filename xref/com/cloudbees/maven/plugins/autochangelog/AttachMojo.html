176     public void execute() throws MojoExecutionException, MojoFailureException {
177         if (skip) {
178             getLog().info("Execution is skipped.");
179             return;
180         }
181         if (previousVersion == null) {
182             getLog().debug(
183                     "Looking for previous release of " + project.getGroupId() + ":" + project.getArtifactId() + ":"
184                             + project.getVersion());
185             Artifact projectArtifact = artifactFactory
186                     .createProjectArtifact(project.getGroupId(), project.getArtifactId(), project.getVersion());
187             ArtifactVersion projectVersion = new DefaultArtifactVersion(project.getVersion());
188 
189             ArtifactVersion latest = null;
190             try {
191                 List<ArtifactVersion> artifactVersions = artifactMetadataSource
192                         .retrieveAvailableVersions(projectArtifact, localRepository,
193                                 project.getRemoteArtifactRepositories());
194                 for (ArtifactVersion version : artifactVersions) {
195                     if (SNAPSHOT_PATTERN.matcher(version.toString()).find() || projectVersion.compareTo(version) <= 0) {
196                         continue;
197                     }
198                     if (latest == null || latest.compareTo(version) < 0) {
199                         latest = version;
200                     }
201                 }
202             } catch (ArtifactMetadataRetrievalException e) {
203                 throw new MojoExecutionException(e.getMessage(), e);
204             }
205             getLog().debug("Previous release = " + latest);
206             previousVersion = latest == null ? null : latest.toString();
207         }
208 
209         Map<GroupArtifactId, String> addedDeps = new LinkedHashMap<GroupArtifactId, String>();
210         Map<GroupArtifactId, String> removedDeps = new LinkedHashMap<GroupArtifactId, String>();
211         Map<GroupArtifactId, String> unchangedDeps = new LinkedHashMap<GroupArtifactId, String>();
212         Map<GroupArtifactId, Map.Entry<String, String>> updatedDeps =
213                 new LinkedHashMap<GroupArtifactId, Map.Entry<String, String>>();
214 
215         for (Dependency dep : project.getDependencies()) {
216             addedDeps.put(new GroupArtifactId(dep), dep.getVersion());
217         }
218 
219         String startTag = null;
220         String endTag = null;
221         if (previousVersion != null) {
222             Artifact latestArtifact = artifactFactory
223                     .createProjectArtifact(project.getGroupId(), project.getArtifactId(), previousVersion);
224             try {
225                 artifactResolver.resolve(latestArtifact, project.getRemoteArtifactRepositories(), localRepository);
226             } catch (ArtifactResolutionException e) {
227                 throw new MojoExecutionException(e.getMessage(), e);
228             } catch (ArtifactNotFoundException e) {
229                 throw new MojoExecutionException(e.getMessage(), e);
230             }
231 
232             ProjectBuildingRequest request = new DefaultProjectBuildingRequest();
233 
234             request.setProcessPlugins(true);
235             request.setProfiles(request.getProfiles());
236             request.setActiveProfileIds(session.getRequest().getActiveProfiles());
237             request.setInactiveProfileIds(session.getRequest().getInactiveProfiles());
238             request.setRemoteRepositories(session.getRequest().getRemoteRepositories());
239             request.setSystemProperties(session.getSystemProperties());
240             request.setUserProperties(session.getUserProperties());
241             request.setRemoteRepositories(session.getRequest().getRemoteRepositories());
242             request.setPluginArtifactRepositories(session.getRequest().getPluginArtifactRepositories());
243             try {
244                 Method getRepositorySession = session.getClass().getMethod("getRepositorySession");
245                 Method setRepositorySession =
246                         request.getClass().getMethod("setRepositorySession", getRepositorySession.getReturnType());
247                 setRepositorySession.invoke(request, getRepositorySession.invoke(session));
248             } catch (NoSuchMethodException e) {
249                 getLog().debug(e);
250             } catch (IllegalAccessException e) {
251                 getLog().debug(e);
252             } catch (InvocationTargetException e) {
253                 getLog().debug(e);
254             }
255             request.setLocalRepository(localRepository);
256             request.setBuildStartTime(session.getRequest().getStartTime());
257             request.setResolveDependencies(true);
258             request.setValidationLevel(ModelBuildingRequest.VALIDATION_LEVEL_STRICT);
259             MavenProject latestProject;
260             try {
261                 latestProject = projectBuilder.build(latestArtifact.getFile(), request).getProject();
262             } catch (ProjectBuildingException e) {
263                 throw new MojoExecutionException(e.getMessage(), e);
264             }
265             for (Dependency dep : latestProject.getDependencies()) {
266                 GroupArtifactId key = new GroupArtifactId(dep);
267                 if (addedDeps.containsKey(key)) {
268                     String v = addedDeps.get(key);
269                     addedDeps.remove(key);
270                     if (!v.equals(dep.getVersion())) {
271                         updatedDeps.put(key, new AbstractMap.SimpleImmutableEntry<String, String>(dep.getVersion(), v));
272                     } else {
273                         unchangedDeps.put(key, v);
274                     }
275                 } else if (updatedDeps.containsKey(key)) {
276                     String v = updatedDeps.get(key).getValue();
277                     if (!v.equals(dep.getVersion())) {
278                         updatedDeps.remove(key);
279                         addedDeps.put(key, v);
280                     } else {
281                         updatedDeps.put(key, new AbstractMap.SimpleImmutableEntry<String, String>(dep.getVersion(), v));
282                     }
283                 } else {
284                     removedDeps.put(key, dep.getVersion());
285                 }
286             }
287 
288             String tagNameFormat = getTagNameFormat(latestProject);
289             getLog().debug("Start Tag name format = " + tagNameFormat);
290             Interpolator interpolator = new StringSearchInterpolator("@{", "}");
291             List<String> possiblePrefixes = java.util.Arrays.asList("project", "pom");
292             Properties values = new Properties();
293             values.setProperty("artifactId", project.getArtifactId());
294             values.setProperty("groupId", project.getGroupId());
295             values.setProperty("version", previousVersion);
296             interpolator.addValueSource(new PrefixedPropertiesValueSource(possiblePrefixes, values, true));
297             RecursionInterceptor recursionInterceptor = new PrefixAwareRecursionInterceptor(possiblePrefixes);
298             try {
299                 startTag = interpolator.interpolate(tagNameFormat, recursionInterceptor);
300             } catch (InterpolationException e) {
301                 throw new MojoExecutionException("Could not interpolate specified tag name format: " + tagNameFormat,
302                         e);
303             }
304         }
305         if (!SNAPSHOT_PATTERN.matcher(project.getVersion()).find()) {
306             String tagNameFormat = getTagNameFormat(project);
307             getLog().debug("End Tag name format = " + tagNameFormat);
308             Interpolator interpolator = new StringSearchInterpolator("@{", "}");
309             List<String> possiblePrefixes = java.util.Arrays.asList("project", "pom");
310             Properties values = new Properties();
311             values.setProperty("artifactId", project.getArtifactId());
312             values.setProperty("groupId", project.getGroupId());
313             values.setProperty("version", project.getVersion());
314             interpolator.addValueSource(new PrefixedPropertiesValueSource(possiblePrefixes, values, true));
315             RecursionInterceptor recursionInterceptor = new PrefixAwareRecursionInterceptor(possiblePrefixes);
316             try {
317                 endTag = interpolator.interpolate(tagNameFormat, recursionInterceptor);
318             } catch (InterpolationException e) {
319                 throw new MojoExecutionException("Could not interpolate specified tag name format: " + tagNameFormat,
320                         e);
321             }
322         }
323         getLog().info("Start tag = " + startTag);
324         getLog().info("End tag = " + endTag);
325 
326         ChangeLogSet changeLog = null;
327 
328         ScmRepository repository;
329         try {
330             repository = getScmRepository();
331             if (repository == null) {
332                 getLog().warn("No SCM connection details, cannot capture source control changes");
333             } else {
334                 ScmProvider provider = null;
335                 try {
336                     provider = manager.getProviderByRepository(repository);
337                 } catch (NoSuchScmProviderException e) {
338                     throw new MojoExecutionException("Unknown/unsupported SCM provider", e);
339                 }
340 
341                 if (repository.getProvider().equals("svn")) {
342                     getLog().warn("SVN does not support the required changelog operations");
343                 } else {
344                     ChangeLogScmRequest changeLogScmRequest =
345                             new ChangeLogScmRequest(repository, new ScmFileSet(basedir));
346                     if (startTag != null) {
347                         changeLogScmRequest.setStartRevision(new ScmTag(startTag));
348                     }
349                     if (endTag != null) {
350                         changeLogScmRequest.setEndRevision(new ScmTag(endTag));
351                     }
352                     ChangeLogScmResult changeLogScmResult;
353                     try {
354                         changeLogScmResult = provider.changeLog(changeLogScmRequest);
355                     } catch (ScmException e) {
356                         // this may just be the end-tag not created yet
357                         if (endTag != null) {
358                             changeLogScmRequest.setStartRevision(new ScmTag(endTag));
359                             try {
360                                 provider.changeLog(changeLogScmRequest);
361                                 // ok the end tag exists, so this looks like a genuine problem
362                                 throw new MojoExecutionException("Could not fetch changelog", e);
363                             } catch (ScmException expected) {
364                                 if (startTag != null) {
365                                     changeLogScmRequest.setStartRevision(new ScmTag(startTag));
366                                 }
367                                 changeLogScmRequest.setEndRevision(null);
368                                 try {
369                                     changeLogScmResult = provider.changeLog(changeLogScmRequest);
370                                 } catch (ScmException issue) {
371                                     throw new MojoExecutionException("Could not fetch changelog", e);
372                                 }
373                             }
374 
375                         } else {
376                             throw new MojoExecutionException("Could not fetch changelog", e);
377                         }
378                     }
379                     checkResult(changeLogScmResult);
380 
381                     changeLog = changeLogScmResult.getChangeLog();
382                 }
383             }
384         } catch (ScmException e) {
385             getLog().error("Could not get SCM repository", e);
386         }
387 
388         if (StringUtils.isEmpty(encoding)) {
389             getLog().warn("Output file encoding has not been set, using platform encoding " + System
390                     .getProperty("file.encoding") + ", i.e. build is platform dependent!");
391             encoding = System.getProperty("file.encoding");
392         }
393 
394         if (!outputDirectory.isDirectory() && !outputDirectory.mkdirs()) {
395             throw new MojoExecutionException("Could not create directory " + outputDirectory);
396         }
397 
398         File outputFile = new File(outputDirectory, finalName + "-changes.json");
399 
400         JsonFactory factory = new JsonFactory();
401         try {
402             JsonGenerator generator = factory.createJsonGenerator(outputFile, JsonEncoding.UTF8);
403             generator.setPrettyPrinter(new DefaultPrettyPrinter());
404             try {
405                 generator.writeStartObject();
406                 generator.writeStringField("previousVersion", previousVersion);
407                 generator.writeFieldName("dependencies");
408                 writeDependenciesDelta(addedDeps, removedDeps, updatedDeps, unchangedDeps, generator);
409                 if (changeLog != null) {
410                     generator.writeFieldName("scm");
411                     writeChangeLog(generator, changeLog);
412                 }
413                 generator.writeEndObject();
414             } finally {
415                 generator.close();
416             }
417         } catch (IOException e) {
418             throw new MojoExecutionException(e.getMessage(), e);
419         }
420         projectHelper.attachArtifact(project, "json", "changes", outputFile);
421     }

423     private void writeDependenciesDelta(Map<GroupArtifactId, String> addedDeps,
424                                         Map<GroupArtifactId, String> removedDeps,
425                                         Map<GroupArtifactId, Map.Entry<String, String>> updatedDeps,
426                                         Map<GroupArtifactId, String> unchangedDeps,
427                                         JsonGenerator generator) throws IOException {
428         generator.writeStartObject();
429         if (!addedDeps.isEmpty()) {
430             generator.writeArrayFieldStart("added");
431             for (Map.Entry<GroupArtifactId, String> e : addedDeps.entrySet()) {
432                 generator.writeStartObject();
433                 generator.writeStringField("groupId", e.getKey().getGroupId());
434                 generator.writeStringField("artifactId", e.getKey().getArtifactId());
435                 generator.writeStringField("version", e.getValue());
436                 generator.writeEndObject();
437             }
438             generator.writeEndArray();
439         }
440         if (!removedDeps.isEmpty()) {
441             generator.writeArrayFieldStart("removed");
442             for (Map.Entry<GroupArtifactId, String> e : removedDeps.entrySet()) {
443                 generator.writeStartObject();
444                 generator.writeStringField("groupId", e.getKey().getGroupId());
445                 generator.writeStringField("artifactId", e.getKey().getArtifactId());
446                 generator.writeStringField("version", e.getValue());
447                 generator.writeEndObject();
448             }
449             generator.writeEndArray();
450         }
451         if (!updatedDeps.isEmpty()) {
452             generator.writeArrayFieldStart("updated");
453             for (Map.Entry<GroupArtifactId, Map.Entry<String, String>> e : updatedDeps.entrySet()) {
454                 generator.writeStartObject();
455                 generator.writeStringField("groupId", e.getKey().getGroupId());
456                 generator.writeStringField("artifactId", e.getKey().getArtifactId());
457                 generator.writeStringField("version", e.getValue().getValue());
458                 generator.writeStringField("originalVersion", e.getValue().getKey());
459                 generator.writeEndObject();
460             }
461             generator.writeEndArray();
462         }
463         if (!unchangedDeps.isEmpty()) {
464             generator.writeArrayFieldStart("unchanged");
465             for (Map.Entry<GroupArtifactId, String> e : unchangedDeps.entrySet()) {
466                 generator.writeStartObject();
467                 generator.writeStringField("groupId", e.getKey().getGroupId());
468                 generator.writeStringField("artifactId", e.getKey().getArtifactId());
469                 generator.writeStringField("version", e.getValue());
470                 generator.writeEndObject();
471             }
472             generator.writeEndArray();
473         }
474         generator.writeEndObject();
475     }

477     private String getTagNameFormat(MavenProject project) {
478         Plugin releasePlugin = null;
479         for (Plugin plugin : project.getBuild().getPlugins()) {
480             if ("org.apache.maven.plugins".equals(plugin.getGroupId()) && "maven-release-plugin"
481                     .equals(plugin.getArtifactId())) {
482                 releasePlugin = plugin;
483                 break;
484             }
485         }
486         String tagNameFormat = "@{project.artifactId}-@{project.version}";
487         if (releasePlugin != null) {
488             Xpp3Dom dom = (Xpp3Dom) releasePlugin.getConfiguration();
489             Xpp3Dom tagNameFormatDom = dom.getChild("tagNameFormat");
490             if (tagNameFormatDom != null) {
491                 tagNameFormat = tagNameFormatDom.getValue();
492             }
493         }
494         return tagNameFormat;
495     }

497     private void writeChangeLog(JsonGenerator generator, ChangeLogSet changeLog) throws IOException {
498         generator.writeStartArray();
499         try {
500             for (ChangeSet changeSet : changeLog.getChangeSets()) {
501                 writeChangeSet(generator, changeSet);
502             }
503         } finally {
504             generator.writeEndArray();
505         }
506     }

508     private void writeChangeSet(JsonGenerator generator, ChangeSet changeSet) throws IOException {
509         generator.writeStartObject();
510         try {
511             if (StringUtils.isNotEmpty(changeSet.getAuthor())) {
512                 generator.writeStringField("author", changeSet.getAuthor());
513             }
514             if (StringUtils.isNotEmpty(changeSet.getComment())) {
515                 generator.writeStringField("comment", changeSet.getComment());
516             }
517             if (changeSet.getDate() != null) {
518                 SimpleDateFormat iso8601format = new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ss'Z'");
519                 iso8601format.setTimeZone(TimeZone.getTimeZone("UTC"));
520                 generator.writeStringField("date", iso8601format.format(changeSet.getDate()));
521             }
522             if (changeSet.getParentRevision() != null) {
523                 generator.writeStringField("parentRevision", changeSet.getParentRevision());
524             }
525             if (!changeSet.getMergedRevisions().isEmpty()) {
526                 generator.writeArrayFieldStart("mergedRevisions");
527                 try {
528                     for (String revision : changeSet.getMergedRevisions()) {
529                         generator.writeString(revision);
530                     }
531                 } finally {
532                     generator.writeEndArray();
533                 }
534             }
535             if (!changeSet.getFiles().isEmpty()) {
536                 generator.writeArrayFieldStart("files");
537                 try {
538                     for (ChangeFile file : changeSet.getFiles()) {
539                         writeChangeFile(generator, file);
540                     }
541                 } finally {
542                     generator.writeEndArray();
543                 }
544             }
545         } finally {
546             generator.writeEndObject();
547         }
548     }

550     private void writeChangeFile(JsonGenerator generator, ChangeFile file) throws IOException {
551         generator.writeStartObject();
552         try {
553             if (file.getAction() != null) {
554                 generator.writeStringField("action", file.getAction().toString());
555             }
556             if (file.getName() != null) {
557                 generator.writeStringField("name", file.getName());
558             }
559             if (file.getRevision() != null) {
560                 generator.writeStringField("revision", file.getRevision());
561             }
562             if (file.getOriginalName() != null) {
563                 generator.writeStringField("originalName", file.getOriginalName());
564             }
565             if (file.getOriginalRevision() != null) {
566                 generator.writeStringField("originalRevision", file.getOriginalRevision());
567             }
568         } finally {
569             generator.writeEndObject();
570         }

573     public ScmRepository getScmRepository()
574             throws ScmException {
575         ScmRepository repository;
576 
577         try {
578             repository = manager.makeScmRepository(getConnection());
579         } catch (MojoExecutionException e) {
580             return null;
581         }
582         try {
583 
584             ScmProviderRepository providerRepo = repository.getProviderRepository();
585 
586             if (!StringUtils.isEmpty(username)) {
587                 providerRepo.setUser(username);
588             }
589 
590             if (!StringUtils.isEmpty(password)) {
591                 providerRepo.setPassword(password);
592             }
593 
594             if (repository.getProviderRepository() instanceof ScmProviderRepositoryWithHost) {
595                 ScmProviderRepositoryWithHost repo = (ScmProviderRepositoryWithHost) repository.getProviderRepository();
596 
597                 loadInfosFromSettings(repo);
598 
599                 if (!StringUtils.isEmpty(username)) {
600                     repo.setUser(username);
601                 }
602 
603                 if (!StringUtils.isEmpty(password)) {
604                     repo.setPassword(password);
605                 }
606 
607                 if (!StringUtils.isEmpty(privateKey)) {
608                     repo.setPrivateKey(privateKey);
609                 }
610 
611                 if (!StringUtils.isEmpty(passphrase)) {
612                     repo.setPassphrase(passphrase);
613                 }
614             }
615 
616             if (!StringUtils.isEmpty(tagBase) && repository.getProvider().equals("svn")) {
617                 SvnScmProviderRepository svnRepo = (SvnScmProviderRepository) repository.getProviderRepository();
618 
619                 svnRepo.setTagBase(tagBase);
620             }
621         } catch (Exception e) {
622             throw new ScmException("Can't load the scm provider.", e);
623         }
624 
625         return repository;
626     }

628     /**
629      * Load username password from settings if user has not set them in JVM properties
630      *
631      * @param repo
632      */
633     private void loadInfosFromSettings(ScmProviderRepositoryWithHost repo) {
634         if (username == null || password == null) {
635             String host = repo.getHost();
636 
637             int port = repo.getPort();
638 
639             if (port > 0) {
640                 host += ":" + port;
641             }
642 
643             Server server = this.settings.getServer(host);
644 
645             if (server != null) {
646                 if (username == null) {
647                     username = this.settings.getServer(host).getUsername();
648                 }
649 
650                 if (password == null) {
651                     password = this.settings.getServer(host).getPassword();
652                 }
653 
654                 if (privateKey == null) {
655                     privateKey = this.settings.getServer(host).getPrivateKey();
656                 }
657 
658                 if (passphrase == null) {
659                     passphrase = this.settings.getServer(host).getPassphrase();
660                 }
661             }
662         }
663     }

680     /**
681      * used to retrieve the SCM connection string
682      *
683      * @return the url string used to connect to the SCM
684      * @throws MojoExecutionException when there is insufficient information to retrieve the SCM connection string
685      */
686     protected String getConnection()
687             throws MojoExecutionException {
688         if (StringUtils.isNotBlank(this.connection)) {
689             return connection;
690         }
691 
692         if (project.getScm() == null) {
693             throw new MojoExecutionException("SCM Connection is not set.");
694         }
695 
696         String scmConnection = project.getScm().getConnection();
697         if (StringUtils.isNotEmpty(scmConnection) && "connection".equals(connectionType.toLowerCase())) {
698             connection = scmConnection;
699         }
700 
701         String scmDeveloper = project.getScm().getDeveloperConnection();
702         if (StringUtils.isNotEmpty(scmDeveloper) && "developerconnection".equals(connectionType.toLowerCase())) {
703             connection = scmDeveloper;
704         }
705 
706         if (StringUtils.isEmpty(connection)) {
707             throw new MojoExecutionException("SCM Connection is not set.");
708         }
709 
710         return connection;
711     }

713     /**
714      * Determines the relative path from trunk to tag, and adds this relative path
715      * to the url.
716      *
717      * @param trunkPath - The trunk url
718      * @param tagPath   - The tag base
719      * @param urlPath   - scm.url or scm.connection
720      * @return The url path for the tag.
721      */
722     private String translateUrlPath(String trunkPath, String tagPath, String urlPath) {
723         trunkPath = trunkPath.trim();
724         tagPath = tagPath.trim();
725         //Strip the slash at the end if one is present
726         if (trunkPath.endsWith("/")) {
727             trunkPath = trunkPath.substring(0, trunkPath.length() - 1);
728         }
729         if (tagPath.endsWith("/")) {
730             tagPath = tagPath.substring(0, tagPath.length() - 1);
731         }
732         char[] tagPathChars = trunkPath.toCharArray();
733         char[] trunkPathChars = tagPath.toCharArray();
734         // Find the common path between trunk and tags
735         int i = 0;
736         while ((i < tagPathChars.length) && (i < trunkPathChars.length) && tagPathChars[i] == trunkPathChars[i]) {
737             ++i;
738         }
739         // If there is nothing common between trunk and tags, or the relative
740         // path does not exist in the url, then just return the tag.
741         if (i == 0 || urlPath.indexOf(trunkPath.substring(i)) < 0) {
742             return tagPath;
743         } else {
744             return StringUtils.replace(urlPath, trunkPath.substring(i), tagPath.substring(i));
745         }
746     }

768                 return true;
769             }
770             if (o == null || getClass() != o.getClass()) {
771                 return false;
772             }
773 
774             GroupArtifactId that = (GroupArtifactId) o;

776             if (artifactId != null ? !artifactId.equals(that.artifactId) : that.artifactId != null) {
777                 return false;
778             }
779             if (groupId != null ? !groupId.equals(that.groupId) : that.groupId != null) {
780                 return false;
781             }
782 
783             return true;
784         }

791         }
