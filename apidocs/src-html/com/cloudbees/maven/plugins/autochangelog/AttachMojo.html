170        public void execute() throws MojoExecutionException, MojoFailureException {
171            if (skip) {
172                getLog().info("Execution is skipped.");
173                return;
174            }
175            getLog().debug("Looking for previous release of " + project.getGroupId() + ":" + project.getArtifactId() + ":"
176                    + project.getVersion());
177            Artifact projectArtifact = artifactFactory
178                    .createProjectArtifact(project.getGroupId(), project.getArtifactId(), project.getVersion());
179            ArtifactVersion projectVersion = new DefaultArtifactVersion(project.getVersion());
180    
181            ArtifactVersion latest = null;
182            try {
183                List<ArtifactVersion> artifactVersions = artifactMetadataSource
184                        .retrieveAvailableVersions(projectArtifact, localRepository,
185                                project.getRemoteArtifactRepositories());
186                for (ArtifactVersion version : artifactVersions) {
187                    if (SNAPSHOT_PATTERN.matcher(version.toString()).find() || projectVersion.compareTo(version) <= 0) {
188                        continue;
189                    }
190                    if (latest == null || latest.compareTo(version) < 0) {
191                        latest = version;
192                    }
193                }
194            } catch (ArtifactMetadataRetrievalException e) {
195                throw new MojoExecutionException(e.getMessage(), e);
196            }
197            getLog().debug("Previous release = " + latest);
198    
199            Map<GroupArtifactId, String> addedDeps = new LinkedHashMap<GroupArtifactId, String>();
200            Map<GroupArtifactId, String> removedDeps = new LinkedHashMap<GroupArtifactId, String>();
201            Map<GroupArtifactId, Map.Entry<String, String>> updatedDeps =
202                    new LinkedHashMap<GroupArtifactId, Map.Entry<String, String>>();
203    
204            for (Dependency dep : project.getDependencies()) {
205                addedDeps.put(new GroupArtifactId(dep), dep.getVersion());
206            }
207    
208            String startTag = null;
209            String endTag = null;
210            if (latest != null) {
211                Artifact latestArtifact = artifactFactory
212                        .createProjectArtifact(project.getGroupId(), project.getArtifactId(), latest.toString());
213                try {
214                    artifactResolver.resolve(latestArtifact, project.getRemoteArtifactRepositories(), localRepository);
215                } catch (ArtifactResolutionException e) {
216                    throw new MojoExecutionException(e.getMessage(), e);
217                } catch (ArtifactNotFoundException e) {
218                    throw new MojoExecutionException(e.getMessage(), e);
219                }
220    
221                ProjectBuildingRequest request = new DefaultProjectBuildingRequest();
222    
223                request.setProcessPlugins(true);
224                request.setProfiles(request.getProfiles());
225                request.setActiveProfileIds(session.getRequest().getActiveProfiles());
226                request.setInactiveProfileIds(session.getRequest().getInactiveProfiles());
227                request.setRemoteRepositories(session.getRequest().getRemoteRepositories());
228                request.setSystemProperties(session.getSystemProperties());
229                request.setUserProperties(session.getUserProperties());
230                request.setRemoteRepositories(session.getRequest().getRemoteRepositories());
231                request.setPluginArtifactRepositories(session.getRequest().getPluginArtifactRepositories());
232                request.setRepositorySession(session.getRepositorySession());
233                request.setLocalRepository(localRepository);
234                request.setBuildStartTime(session.getRequest().getStartTime());
235                request.setResolveDependencies(true);
236                request.setValidationLevel(ModelBuildingRequest.VALIDATION_LEVEL_STRICT);
237                MavenProject latestProject;
238                try {
239                    latestProject = projectBuilder.build(latestArtifact.getFile(), request).getProject();
240                } catch (ProjectBuildingException e) {
241                    throw new MojoExecutionException(e.getMessage(), e);
242                }
243                for (Dependency dep : latestProject.getDependencies()) {
244                    GroupArtifactId key = new GroupArtifactId(dep);
245                    if (addedDeps.containsKey(key)) {
246                        String v = addedDeps.get(key);
247                        addedDeps.remove(key);
248                        if (!v.equals(dep.getVersion())) {
249                            updatedDeps.put(key, new AbstractMap.SimpleImmutableEntry<String, String>(dep.getVersion(), v));
250                        }
251                    } else if (updatedDeps.containsKey(key)) {
252                        String v = updatedDeps.get(key).getValue();
253                        if (!v.equals(dep.getVersion())) {
254                            updatedDeps.remove(key);
255                            addedDeps.put(key, v);
256                        } else {
257                            updatedDeps.put(key, new AbstractMap.SimpleImmutableEntry<String, String>(dep.getVersion(), v));
258                        }
259                    } else {
260                        removedDeps.put(key, dep.getVersion());
261                    }
262                }
263    
264                String tagNameFormat = getTagNameFormat(latestProject);
265                getLog().debug("Start Tag name format = " + tagNameFormat);
266                Interpolator interpolator = new StringSearchInterpolator("@{", "}");
267                List<String> possiblePrefixes = java.util.Arrays.asList("project", "pom");
268                Properties values = new Properties();
269                values.setProperty("artifactId", project.getArtifactId());
270                values.setProperty("groupId", project.getGroupId());
271                values.setProperty("version", latest.toString());
272                interpolator.addValueSource(new PrefixedPropertiesValueSource(possiblePrefixes, values, true));
273                RecursionInterceptor recursionInterceptor = new PrefixAwareRecursionInterceptor(possiblePrefixes);
274                try {
275                    startTag = interpolator.interpolate(tagNameFormat, recursionInterceptor);
276                } catch (InterpolationException e) {
277                    throw new MojoExecutionException("Could not interpolate specified tag name format: " + tagNameFormat,
278                            e);
279                }
280            }
281            if (!SNAPSHOT_PATTERN.matcher(project.getVersion()).find()) {
282                String tagNameFormat = getTagNameFormat(project);
283                getLog().debug("End Tag name format = " + tagNameFormat);
284                Interpolator interpolator = new StringSearchInterpolator("@{", "}");
285                List<String> possiblePrefixes = java.util.Arrays.asList("project", "pom");
286                Properties values = new Properties();
287                values.setProperty("artifactId", project.getArtifactId());
288                values.setProperty("groupId", project.getGroupId());
289                values.setProperty("version", project.getVersion());
290                interpolator.addValueSource(new PrefixedPropertiesValueSource(possiblePrefixes, values, true));
291                RecursionInterceptor recursionInterceptor = new PrefixAwareRecursionInterceptor(possiblePrefixes);
292                try {
293                    endTag = interpolator.interpolate(tagNameFormat, recursionInterceptor);
294                } catch (InterpolationException e) {
295                    throw new MojoExecutionException("Could not interpolate specified tag name format: " + tagNameFormat,
296                            e);
297                }
298            }
299            getLog().info("Start tag = " + startTag);
300            getLog().info("End tag = " + endTag);
301    
302            ChangeLogSet changeLog = null;
303    
304            ScmRepository repository;
305            try {
306                repository = getScmRepository();
307                ScmProvider provider = null;
308                try {
309                    provider = manager.getProviderByRepository(repository);
310                } catch (NoSuchScmProviderException e) {
311                    throw new MojoExecutionException("Unknown/unsupported SCM provider", e);
312                }
313    
314                ScmVersion scmStartVersion;
315                ScmVersion scmEndVersion;
316                if (repository.getProvider().equals("svn")) {
317                    getLog().warn("SVN does not support the required changelog operations");
318                } else {
319                    scmStartVersion = startTag == null ? null : new ScmTag(startTag);
320                    scmEndVersion = endTag == null ? null : new ScmTag(endTag);
321                    ChangeLogScmResult changeLogScmResult;
322                    try {
323                        changeLogScmResult =
324                                provider.changeLog(repository, new ScmFileSet(basedir), scmStartVersion, scmEndVersion);
325                    } catch (ScmException e) {
326                        throw new MojoExecutionException("Could not fetch changelog", e);
327                    }
328                    checkResult(changeLogScmResult);
329    
330                    changeLog = changeLogScmResult.getChangeLog();
331                }
332            } catch (ScmException e) {
333                getLog().error("Could not get SCM repository", e);
334            }
335    
336            if (StringUtils.isEmpty(encoding)) {
337                getLog().warn("Output file encoding has not been set, using platform encoding " + System
338                        .getProperty("file.encoding") + ", i.e. build is platform dependent!");
339                encoding = System.getProperty("file.encoding");
340            }
341    
342            if (!outputDirectory.isDirectory() && !outputDirectory.mkdirs()) {
343                throw new MojoExecutionException("Could not create directory " + outputDirectory);
344            }
345    
346            File outputFile = new File(outputDirectory, finalName + "-changes.json");
347    
348            JsonFactory factory = new JsonFactory();
349            try {
350                JsonGenerator generator = factory.createJsonGenerator(outputFile, JsonEncoding.UTF8);
351                generator.setPrettyPrinter(new DefaultPrettyPrinter());
352                try {
353                    generator.writeStartObject();
354                    generator.writeFieldName("dependencies");
355                    writeDependenciesDelta(addedDeps, removedDeps, updatedDeps, generator);
356                    if (changeLog != null) {
357                        generator.writeFieldName("scm");
358                        writeChangeLog(generator, changeLog);
359                    }
360                    generator.writeEndObject();
361                } finally {
362                    generator.close();
363                }
364            } catch (IOException e) {
365                throw new MojoExecutionException(e.getMessage(), e);
366            }
367            projectHelper.attachArtifact(project, "json", "changes", outputFile);
368        }

370        private void writeDependenciesDelta(Map<GroupArtifactId, String> addedDeps,
371                                            Map<GroupArtifactId, String> removedDeps,
372                                            Map<GroupArtifactId, Map.Entry<String, String>> updatedDeps,
373                                            JsonGenerator generator) throws IOException {
374            generator.writeStartObject();
375            if (!addedDeps.isEmpty()) {
376                generator.writeArrayFieldStart("added");
377                for (Map.Entry<GroupArtifactId, String> e : addedDeps.entrySet()) {
378                    generator.writeStartObject();
379                    generator.writeStringField("groupId", e.getKey().getGroupId());
380                    generator.writeStringField("artifactId", e.getKey().getArtifactId());
381                    generator.writeStringField("version", e.getValue());
382                    generator.writeEndObject();
383                }
384                generator.writeEndArray();
385            }
386            if (!removedDeps.isEmpty()) {
387                generator.writeArrayFieldStart("removed");
388                for (Map.Entry<GroupArtifactId, String> e : removedDeps.entrySet()) {
389                    generator.writeStartObject();
390                    generator.writeStringField("groupId", e.getKey().getGroupId());
391                    generator.writeStringField("artifactId", e.getKey().getArtifactId());
392                    generator.writeStringField("version", e.getValue());
393                    generator.writeEndObject();
394                }
395                generator.writeEndArray();
396            }
397            if (!updatedDeps.isEmpty()) {
398                generator.writeArrayFieldStart("updated");
399                for (Map.Entry<GroupArtifactId, Map.Entry<String, String>> e : updatedDeps.entrySet()) {
400                    generator.writeStartObject();
401                    generator.writeStringField("groupId", e.getKey().getGroupId());
402                    generator.writeStringField("artifactId", e.getKey().getArtifactId());
403                    generator.writeStringField("version", e.getValue().getValue());
404                    generator.writeStringField("originalVersion", e.getValue().getKey());
405                    generator.writeEndObject();
406                }
407                generator.writeEndArray();
408            }
409            generator.writeEndObject();
410        }

412        private String getTagNameFormat(MavenProject project) {
413            Plugin releasePlugin = null;
414            for (Plugin plugin : project.getBuild().getPlugins()) {
415                if ("org.apache.maven.plugins".equals(plugin.getGroupId()) && "maven-release-plugin"
416                        .equals(plugin.getArtifactId())) {
417                    releasePlugin = plugin;
418                    break;
419                }
420            }
421            String tagNameFormat = "@{project.artifactId}-@{project.version}";
422            if (releasePlugin != null) {
423                Xpp3Dom dom = (Xpp3Dom) releasePlugin.getConfiguration();
424                Xpp3Dom tagNameFormatDom = dom.getChild("tagNameFormat");
425                if (tagNameFormatDom != null) {
426                    tagNameFormat = tagNameFormatDom.getValue();
427                }
428            }
429            return tagNameFormat;
430        }

432        private void writeChangeLog(JsonGenerator generator, ChangeLogSet changeLog) throws IOException {
433            generator.writeStartArray();
434            try {
435                for (ChangeSet changeSet : changeLog.getChangeSets()) {
436                    writeChangeSet(generator, changeSet);
437                }
438            } finally {
439                generator.writeEndArray();
440            }
441        }
442    
443        private void writeChangeSet(JsonGenerator generator, ChangeSet changeSet) throws IOException {
444            generator.writeStartObject();
445            try {
446                if (StringUtils.isNotEmpty(changeSet.getAuthor())) {
447                    generator.writeStringField("author", changeSet.getAuthor());
448                }
449                if (StringUtils.isNotEmpty(changeSet.getComment())) {
450                    generator.writeStringField("comment", changeSet.getComment());
451                }
452                if (changeSet.getDate() != null) {
453                    SimpleDateFormat iso8601format = new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ss'Z'");
454                    iso8601format.setTimeZone(TimeZone.getTimeZone("UTC"));
455                    generator.writeStringField("date", iso8601format.format(changeSet.getDate()));
456                }
457                if (changeSet.getParentRevision() != null) {
458                    generator.writeStringField("parentRevision", changeSet.getParentRevision());
459                }
460                if (!changeSet.getMergedRevisions().isEmpty()) {
461                    generator.writeArrayFieldStart("mergedRevisions");
462                    try {
463                        for (String revision : changeSet.getMergedRevisions()) {
464                            generator.writeString(revision);
465                        }
466                    } finally {
467                        generator.writeEndArray();
468                    }
469                }
470                if (!changeSet.getFiles().isEmpty()) {
471                    generator.writeArrayFieldStart("files");
472                    try {
473                        for (ChangeFile file : changeSet.getFiles()) {
474                            writeChangeFile(generator, file);
475                        }
476                    } finally {
477                        generator.writeEndArray();
478                    }
479                }
480            } finally {
481                generator.writeEndObject();
482            }
483        }
484    
485        private void writeChangeFile(JsonGenerator generator, ChangeFile file) throws IOException {
486            generator.writeStartObject();
487            try {
488                if (file.getAction() != null) {
489                    generator.writeStringField("action", file.getAction().toString());
490                }
491                if (file.getName() != null) {
492                    generator.writeStringField("name", file.getName());
493                }
494                if (file.getRevision() != null) {
495                    generator.writeStringField("revision", file.getRevision());
496                }
497                if (file.getOriginalName() != null) {
498                    generator.writeStringField("originalName", file.getOriginalName());
499                }
500                if (file.getOriginalRevision() != null) {
501                    generator.writeStringField("originalRevision", file.getOriginalRevision());
502                }
503            } finally {
504                generator.writeEndObject();
505            }
506        }

508        public ScmRepository getScmRepository()
509                throws ScmException {
510            ScmRepository repository;
511    
512            try {
513                repository = manager.makeScmRepository(getConnection());
514    
515                ScmProviderRepository providerRepo = repository.getProviderRepository();
516    
517                if (!StringUtils.isEmpty(username)) {
518                    providerRepo.setUser(username);
519                }
520    
521                if (!StringUtils.isEmpty(password)) {
522                    providerRepo.setPassword(password);
523                }
524    
525                if (repository.getProviderRepository() instanceof ScmProviderRepositoryWithHost) {
526                    ScmProviderRepositoryWithHost repo = (ScmProviderRepositoryWithHost) repository.getProviderRepository();
527    
528                    loadInfosFromSettings(repo);
529    
530                    if (!StringUtils.isEmpty(username)) {
531                        repo.setUser(username);
532                    }
533    
534                    if (!StringUtils.isEmpty(password)) {
535                        repo.setPassword(password);
536                    }
537    
538                    if (!StringUtils.isEmpty(privateKey)) {
539                        repo.setPrivateKey(privateKey);
540                    }
541    
542                    if (!StringUtils.isEmpty(passphrase)) {
543                        repo.setPassphrase(passphrase);
544                    }
545                }
546    
547                if (!StringUtils.isEmpty(tagBase) && repository.getProvider().equals("svn")) {
548                    SvnScmProviderRepository svnRepo = (SvnScmProviderRepository) repository.getProviderRepository();
549    
550                    svnRepo.setTagBase(tagBase);
551                }
552            } catch (Exception e) {
553                throw new ScmException("Can't load the scm provider.", e);
554            }
555    
556            return repository;
557        }

559        /**
560         * Load username password from settings if user has not set them in JVM properties
561         *
562         * @param repo
563         */
564        private void loadInfosFromSettings(ScmProviderRepositoryWithHost repo) {
565            if (username == null || password == null) {
566                String host = repo.getHost();
567    
568                int port = repo.getPort();
569    
570                if (port > 0) {
571                    host += ":" + port;
572                }
573    
574                Server server = this.settings.getServer(host);
575    
576                if (server != null) {
577                    if (username == null) {
578                        username = this.settings.getServer(host).getUsername();
579                    }
580    
581                    if (password == null) {
582                        password = this.settings.getServer(host).getPassword();
583                    }
584    
585                    if (privateKey == null) {
586                        privateKey = this.settings.getServer(host).getPrivateKey();
587                    }
588    
589                    if (passphrase == null) {
590                        passphrase = this.settings.getServer(host).getPassphrase();
591                    }
592                }
593            }
594        }

596        public void checkResult(ScmResult result)
597                throws MojoExecutionException {
598            if (!result.isSuccess()) {
599                getLog().error("Provider message:");
600    
601                getLog().error(result.getProviderMessage() == null ? "" : result.getProviderMessage());
602    
603                getLog().error("Command output:");
604    
605                getLog().error(result.getCommandOutput() == null ? "" : result.getCommandOutput());
606    
607                throw new MojoExecutionException("Command failed.");
608            }
609        }

611        /**
612         * used to retrieve the SCM connection string
613         *
614         * @return the url string used to connect to the SCM
615         * @throws MojoExecutionException when there is insufficient information to retrieve the SCM connection string
616         */
617        protected String getConnection()
618                throws MojoExecutionException {
619            if (this.connection != null) {
620                return connection;
621            }
622    
623            if (project.getScm() == null) {
624                throw new MojoExecutionException("SCM Connection is not set.");
625            }
626    
627            String scmConnection = project.getScm().getConnection();
628            if (StringUtils.isNotEmpty(scmConnection) && "connection".equals(connectionType.toLowerCase())) {
629                connection = scmConnection;
630            }
631    
632            String scmDeveloper = project.getScm().getDeveloperConnection();
633            if (StringUtils.isNotEmpty(scmDeveloper) && "developerconnection".equals(connectionType.toLowerCase())) {
634                connection = scmDeveloper;
635            }
636    
637            if (StringUtils.isEmpty(connection)) {
638                throw new MojoExecutionException("SCM Connection is not set.");
639            }
640    
641            return connection;
642        }

644        /**
645         * Determines the relative path from trunk to tag, and adds this relative path
646         * to the url.
647         *
648         * @param trunkPath - The trunk url
649         * @param tagPath   - The tag base
650         * @param urlPath   - scm.url or scm.connection
651         * @return The url path for the tag.
652         */
653        private String translateUrlPath(String trunkPath, String tagPath, String urlPath) {
654            trunkPath = trunkPath.trim();
655            tagPath = tagPath.trim();
656            //Strip the slash at the end if one is present
657            if (trunkPath.endsWith("/")) {
658                trunkPath = trunkPath.substring(0, trunkPath.length() - 1);
659            }
660            if (tagPath.endsWith("/")) {
661                tagPath = tagPath.substring(0, tagPath.length() - 1);
662            }
663            char[] tagPathChars = trunkPath.toCharArray();
664            char[] trunkPathChars = tagPath.toCharArray();
665            // Find the common path between trunk and tags
666            int i = 0;
667            while ((i < tagPathChars.length) && (i < trunkPathChars.length) && tagPathChars[i] == trunkPathChars[i]) {
668                ++i;
669            }
670            // If there is nothing common between trunk and tags, or the relative
671            // path does not exist in the url, then just return the tag.
672            if (i == 0 || urlPath.indexOf(trunkPath.substring(i)) < 0) {
673                return tagPath;
674            } else {
675                return StringUtils.replace(urlPath, trunkPath.substring(i), tagPath.substring(i));
676            }
677        }

679        private static class GroupArtifactId {
680            private final String groupId;
681            private final String artifactId;
682    
683            public GroupArtifactId(Dependency dependency) {
684                this.groupId = dependency.getGroupId();
685                this.artifactId = dependency.getArtifactId();
686            }
687    
688            public String getArtifactId() {
689                return artifactId;
690            }
691    
692            public String getGroupId() {
693                return groupId;
694            }
695    
696            @Override
697            public boolean equals(Object o) {
698                if (this == o) {
699                    return true;
700                }
701                if (o == null || getClass() != o.getClass()) {
702                    return false;
703                }
704    
705                GroupArtifactId that = (GroupArtifactId) o;
706    
707                if (artifactId != null ? !artifactId.equals(that.artifactId) : that.artifactId != null) {
708                    return false;
709                }
710                if (groupId != null ? !groupId.equals(that.groupId) : that.groupId != null) {
711                    return false;
712                }
713    
714                return true;
715            }
716    
717            @Override
718            public int hashCode() {
719                int result = groupId != null ? groupId.hashCode() : 0;
720                result = 31 * result + (artifactId != null ? artifactId.hashCode() : 0);
721                return result;
722            }
723    
724            @Override
725            public String toString() {
726                return getGroupId() + ":" + getArtifactId();
727            }
